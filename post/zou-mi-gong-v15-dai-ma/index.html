#include<iostream>

using namespace std;
char map[100][100];
int N; 
bool vst[100][100];
int dx[4]={0,1,0,-1};
int dy[4]={1,0,-1,0};
void print(){
	for(int i=1;i<=N;i++,cout<<endl)
		for(int j=1;j<=N;j++)
			cout<<map[i][j];
}
void dfs(int x,int y){
	if(map[x][y]=='#'&&x==1&&y==1)return ; 
	vst[x][y]=1;	
	if(vst[N][N])return;
	for(int k=0;k<4;k++){
		int nx=x+dx[k],ny=y+dy[k];
		if(!vst[nx][ny]&&map[nx][ny]=='_'&&nx>=1&&nx<=N&&ny>=1&&ny<=N){						
			dfs(nx,ny);			
		}
	}
}

int main(){
	srand(time(0));
	cout<<"输入wasd控制方向，_为空地，#为墙，*为当前地点,P为终点\n";//介绍 
	cout<<"输入地图大小：";
	cin>>N; 
	cout<<"地图生成中……";
	while(1){				
		for(int i=1;i<=N;i++)
			for(int j=1;j<=N;j++)
			{				
				int a=rand()%8;
				if(a<=3)map[i][j]='#';
				else map[i][j]='_';
			}
		dfs(1,1);
		if(vst[N][N]==1)break;
		for(int i=1;i<=N;i++)for(int j=1;j<=N;j++) vst[i][j]=0;
	}
	system("cls"); 
	map[1][1]='*';
	map[N][N]='P';
	cout<<"选择难度\n1为简单版，2为普通版，3为困难版,4为极难版（伸手不见五指）";
	int nd;
	cin>>nd;
	if(nd==1)
		easy();
	else if(nd==2) 
		normal(2);
	else if(nd==3)
		normal(1);
	else if(nd==4)
		normal(0); 
	else cout<<"gameover!";
	
	system("pause");
	return 0;
}
