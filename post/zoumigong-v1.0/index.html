#include<iostream>
#include<cstdlib>
#include<ctime> 
using namespace std;
char map[100][100];
int N; 
bool vst[100][100];
int dx[4]={0,1,0,-1};
int dy[4]={1,0,-1,0};
void print(){
	for(int i=1;i<=N;i++,cout<<endl)
		for(int j=1;j<=N;j++)
			cout<<map[i][j];
}
void dfs(int x,int y){
	if(map[x][y]=='#'&&x==1&&y==1)return ; 
	vst[x][y]=1;	
	if(vst[N][N])return;
	for(int k=0;k<4;k++){
		int nx=x+dx[k],ny=y+dy[k];
		if(!vst[nx][ny]&&map[nx][ny]=='o'&&nx>=1&&nx<=N&&ny>=1&&ny<=N){						
			dfs(nx,ny);			
		}
	}
}

int main(){
	srand(time(0));
	cout<<"输入wasd控制方向，o为空地，#为墙，*为当前地点,右下角为终点\n";//介绍 
	cout<<"输入地图大小：";
	cin>>N; 
	cout<<"地图生成中……";
	while(1){				
		for(int i=1;i<=N;i++)
			for(int j=1;j<=N;j++)
			{				
				int a=rand()%7;
				if(a<3)map[i][j]='#';
				else map[i][j]='o';
			}
		dfs(1,1);
		if(vst[N][N]==1)break;
		for(int i=1;i<=N;i++)for(int j=1;j<=N;j++) vst[i][j]=0;
	}
	system("cls"); 
	map[1][1]='*';
	print();
	int x=1,y=1;
	while(1){
		if(x==N&&y==N){
			cout<<"到了！\n";
			system("pause");
			return 0; 
		}
		char a;
		cin>>a;
		if(a=='w'){
			int nx=x+dx[3],ny=y+dy[3];
			if(map[nx][ny]=='#'||nx<=0||nx>N||ny<=0||ny>N){
				cout<<"撞墙了QAQ\n";
				continue; 
			}
			else{
				map[x][y]='o';
				map[nx][ny]='*';
				system("cls");
				print();
				x=nx,y=ny;
			}
		}
		else if(a=='d'){//correct
			int nx=x+dx[0],ny=y+dy[0];
			if(map[nx][ny]=='#'||nx<=0||nx>N||ny<=0||ny>N){
				cout<<"撞墙了QAQ\n";
				continue; 
			}
			else{
				map[x][y]='o';
				map[nx][ny]='*';
				system("cls");
				print();
				x=nx,y=ny;
			}
		}
		else if(a=='s'){
			int nx=x+dx[1],ny=y+dy[1];
			if(map[nx][ny]=='#'||nx<=0||nx>N||ny<=0||ny>N){
				cout<<"撞墙了QAQ\n";
				continue; 
			}
			else{
				map[x][y]='o';
				map[nx][ny]='*';
				system("cls");
				print();
				x=nx,y=ny;
			}
		}
		else if(a=='a'){
			int nx=x+dx[2],ny=y+dy[2];
			if(map[nx][ny]=='#'||nx<=0||nx>N||ny<=0||ny>N){
				cout<<"撞墙了QAQ\n";
				continue; 
			}
			else{
				map[x][y]='o';
				map[nx][ny]='*';
				system("cls");
				print();
				x=nx,y=ny;
			}
		}
		else{
			cout<<"大哥您输入了什么鬼啊";
			continue; 
		}
	}
	system("pause");
}
